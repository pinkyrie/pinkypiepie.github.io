  • void MaxHeap<T>::BuildHeap(){
    for(int i=CurrentSize/2-1;i>=0;i--){
        SiftDown(i);
    }
}
void MaxHeap<T>::SiftDown(int left){
    int i=left;
    int j=2*i+1;
    T temp=HeapArray[i];
    while(j<CurrentSize){
        j++;
        if(temp<heapArray[i]){
            heapArray[i]=heapArray[j];
            i=j;
            j=2*j+1;
        }
        else break;
    }
    heapArray[i]=temp;    
}
void MaxHeap<T>::SiftUp(int pos){
    int p=parent(pos);
    while(p>-1&&heapArray[p]<heapArray[pos]){
        temp=heapArray[p];
        heapArray[p]=heapArray[pos];
        heapArray[pos]=temp;
        p=parent(p);
    }
}
